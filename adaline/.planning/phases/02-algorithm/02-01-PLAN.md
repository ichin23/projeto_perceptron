```python
class ADALINE:
    def __init__(self, eta=0.0025, epsilon=1e-6):
        self.eta = eta
        self.epsilon = epsilon
        self.weights = None
        self.mse_history = []

    def train(self, X, d):
        """
        Online Widrow-Hoff (LMS) training.
        X: (N, n_features+1) — bias already prepended as x0=1
        d: (N,) — target labels -1 or +1
        Returns: epochs until convergence
        """
        N, n = X.shape
        self.weights = np.random.uniform(0, 1, n)  # w0...w4
        self.mse_history = []
        prev_mse = None

        for epoch in range(1, 100_001):
            errors = []
            for i in range(N):
                net = np.dot(self.weights, X[i])   # linear activation
                error = d[i] - net                  # delta rule error
                self.weights += self.eta * error * X[i]  # weight update
                errors.append(error)

            mse = np.mean(np.array(errors) ** 2)
            self.mse_history.append(mse)

            if prev_mse is not None and abs(mse - prev_mse) < self.epsilon:
                return epoch
            prev_mse = mse

        return 100_000  # max epochs fallback

    def predict(self, X):
        """
        Classify inputs using sign of linear output.
        Returns: array of -1 or +1
        """
        net = X @ self.weights
        return np.sign(net)

    def net_output(self, X):
        """Raw linear output (before sign)."""
        return X @ self.weights
```
